36iDog::iDog(IWDG_TypeDef* iDogHandle, uint16_t reloadValue, iDog_prescaler_t prescaleValue, iDog_mode_t mode)
37{
38 // Set watchdog handle
39 handle = iDogHandle;
40
41 // Set watchdog prescaler
42 setPrescaler(prescaleValue);
43
44 // Set watchdog reload value
45 setReloadValue(reloadValue);
46
47 // Set watchdog behaviour when core is halted
48 if (mode == DBG_HALT)
49 {
50 // Enable clock for DBGMCU
51 RCC->APB2ENR |= RCC_APB2ENR_DBGEN;
52
53 // Disable IWDG when core is halted
54 DBGMCU->APB1FZ |= DBGMCU_APB1_FZ_DBG_IWDG_STOP;
55
56 // Disable clock for DBGMCU
57 RCC->APB2ENR &= ~RCC_APB2ENR_DBGEN;
58 }
59}

68void iDog::setPrescaler(iDog_prescaler_t value)
69{
70 // Wait if prescaler update is ongoing
71 while (handle->SR & IWDG_SR_PVU);
72
73 // Enable write access
74 access();
75
76 // Write prescaler value
77 handle->PR = value;
78
79 // Feed watchdog
80 feed();
81}

83void iDog::setReloadValue(uint16_t value)
84{
85 // Wait if reload value update is ongoing
86 while (handle->SR & IWDG_SR_RVU);
87
88 // Enable write access
89 access();
90
91 // Make sure value does not exceed value defined with IDOG_MAX_RELOAD
92 if (value > IDOG_MAX_RELOAD) value = IDOG_MAX_RELOAD;
93
94 // Write reload value
95 handle->RLR = value;
96
97 // Feed watchdog
98 feed();
99}
